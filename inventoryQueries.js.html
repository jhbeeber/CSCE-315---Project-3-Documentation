/** 
 * @module queries/inventory
 * @requires express
 */

const express = require('express');
const db = require('../database');
const router = express.Router();

/**
 * Query for all information associated with a specific inventory item using the inventory id number.
 * @module queries/inventory
 */
router.get("/inventory/all", async (req, res) => {
    try {
        const result = await db.query("SELECT * FROM inventory order by inventoryid");
        res.json(result.rows);
        res.status(200);
    } catch (error) {
        console.error("Error querying the database: ", error);
        res.status(500).json({ error: "Internal Server Error" });
    }
});

/**
 * Update the quantity of an inventory item using the inventory id number and inputted quantity.
 * @module queries/inventory
 */
router.put("/inventory/updateQuantity/:invID&:quantity", async (req, res) => {
    const invID = req.params.invID.split('=')[1];
    const quantity = req.params.quantity.split('=')[1];
    try {
        const result = await db.query(`UPDATE inventory SET quantity = ${quantity} where inventoryID = ${invID};`);
        res.status(201).json("Done");
    } catch (error) {
        console.error("Error querying the database: ", error);
        res.status(500).json({ error: "Internal Server Error" });
    }
});

/**
 * Add inventory item using the inventory item name and the quantity.
 * @module queries/inventory
 */
router.put("/inventory/addItem/:itemName&:quantity",  async (req, res) => {
    const itemName = req.params.itemName.split('=')[1];
    const quantity = req.params.quantity.split('=')[1];
    try {
        //Need to get the last inventoryID to create new item
        const idResult = await db.query("SELECT MAX(inventoryID) FROM inventory;");
        const id = idResult.rows[0].max + 1;
        const result = await db.query(`INSERT INTO inventory (inventoryID, name, quantity) VALUES (${id}, '${itemName}', ${quantity});`);
        res.status(201).json("Done");
    } catch (error) {
        console.error("Error querying the database: ", error);
        res.status(500).json({ error: "Internal Server Error" });
    }
});

 * Delete all occurrences of an inventory item from the invenotry relationship table.
