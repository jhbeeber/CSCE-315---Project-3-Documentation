/**
 * Gets the current weather based on the user's geolocation. If the user doesn't grant access, provides geolocation isn't supported message.
 * @function
 */
function getWeather() {
    if (navigator.geolocation) {
        navigator.geolocation.getCurrentPosition(function(position) {
            const latitude = position.coords.latitude;
            const longitude = position.coords.longitude;

            fetch("https://api.open-meteo.com/v1/forecast?latitude=" + latitude + "&longitude=" + longitude + "&current=temperature_2m,relative_humidity_2m,is_day,precipitation&temperature_unit=fahrenheit&precipitation_unit=inch")
            .then((response) => response.json())
            .then((data) => {
                const itemDiv = document.getElementById("weatherDiv");
                itemDiv.innerHTML = '<p aria-label="Temperature">' + data.current.temperature_2m + ' F</p>' + 
                '<p aria-label="Humidity">' + data.current.relative_humidity_2m + '% humidity</p>' + 
                '<p aria-label="Precipitation">' + data.current.precipitation + ' inches of precipitation</p>';
                if(data.current.is_day) {
                    itemDiv.style.backgroundColor = "skyblue";
                    itemDiv.style.color = "black";
                } else {
                    itemDiv.style.backgroundColor = "darkblue";
                    itemDiv.style.color = "white";
                }
            })
            .catch((error) => {
                console.error("Error fetching data:", error);
            });
        });
    } else {
        console.log("Geolocation is not supported by this browser.");
        const itemDiv = document.getElementById("weatherDiv");
        itemDiv.innerHTML = "<p>Weather unavailable.<br>Geolocation services not permitted.</p>";
    }
}
getWeather();
